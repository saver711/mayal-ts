بداية الشركة

نشـأت الشـركة منـذ 1413/11/27هــ - 1993/5/19م وخـال هـذه الفتـرة اسـتطعنا إثبـات جدارتنـا ومقدرتنـا مـن خــال عمــل احترافــي مبنــي علــى الجــودة والســرعة. بــدأت الشــركة أعمالهــا فــي مدينــة جــدة وامتــدت فروعهــا إلــى الريـاض والمدينـة المنـورة ومكـة المكرمـة وينبـع و مهـد الدهـب ونسـعى للتوسـع فـي جميـع أنحـاء المملكـة وخـال هـذه الفتــرة قامــت بتنفيــذ وإنشــاء وتشــغيل وصيانــة مشــاريع الأعمــال الكهربائيــة الخاصــة لشــبكات الجهــد العالــي والمتوسـط والمنخفـض والمسـاهمة فـي مجـال التطويـر العقـاري والاسـتثمار لتحقـق بذلـك نقلـة نوعيـة فـي طريقـة عملهـا مـن حيـث الحجـم و أسـلوب العمـل .

كلمة المدير العام والمؤسس

المدير العام
نايف عبدالله صالح رقبان

تتواصل مسيرة العمل والإنجازات على امتداد ارض المملكة الحبيبة وتتسارع الخطى وتتعاظم الخدمات وتتوارى الإنجازات للتوسـع الملحـوظ ً فـي مواقـع العمـل والتنميـة ، ومملكتنـا الحبيبـة تخطـو بـكل ثقـة وثبـات نحـو مصـاف الـدول المتقدمـة. ونظـرا فـي سـوق الاسـتثمار العقـاري ً والمقـاولات فـي المملكـة كان لزامـا علينـا فـي شـركة الميـال للمقـاولات أن يكـون لنـا بصمـة واضحـة فـي هـذا المجـال الواسـع فاسـتطعنا بعـون الله وتوفيقـه خـال فتـرة عملنـا أن ننجـز العديـد مـن المشـاريع المتنوعـة واضعيـن هدفنـا الأساســي فــي تقديــم حلــول مثاليــة لتقديــم الخدمــة الكهربائيــة مــن خــال انشــاء وصيانــة وتحســين شــبكات النقــل والتوزيــع الكهربائيـة وايصـال الطاقـة الكهربائيـة للمشـتركين ، وتنفيـذ المشـاريع للقطاعيـن الحكومـي والخـاص مـن خـال حلـول مبتكـره بـكل دقـه واحترافيـة فـي العمـل وفـق اعلـى المقاييـس والمواصفـات العالميـة

المؤسس
عبدالله صالح سعيد رقبان

منـذ تأسيسـنا شـركة الميـال للمقـاولات وشـعارنا هـو الجـودة والالتـزام وهمـا الجناحـان اللـذان حلقـت بهمـا الشـركة فـي سـماء الاقتصــاد الواســع بالمملكــة العربيــة الســعودية وســط أجــواء تنافســية غيــر اعتياديــة فكانــت تلــك الرؤيــة المتقدمــة بمثابــة الجــواد الأصيــل الــذي رســمت آثــار خطــاه علــي أرض الوطــن بإنجــازات قيمــة شــهدت بالوفــاء برؤيتنــا فــي القطاعيــن العــام والخـاص ونطمـح بـكل ثقـة فـي الدفـع بالشـركة إلـي مصـاف الشـركات الوطنيـة العماقـة. فـي مجـال الاقتصـاد الواسـع وفـي قطـاع تنافســي وعمــاق كقطــاع المقــاولات بالمملكــة العربيــة الســعودية فــإن روح المثابــرة والتحــدي لهــي الحــراك الرئيســي لتحقيــق الأهـداف الصعبـة والمتميـزة .كمـا أن الرؤيـة المتقدمـة التـي تعتمـد علـى مـوارد بشـرية أكثـر فاعليـة ونظـم تقنيـة حديثـة كانـت خطتنــا للســير بخطــى ثابتــة وســريعة ومواكبــة للمســتجدات الاقتصاديــة فــي الســوق الداخلــي وإن فريــق عمــل شــركة الميــال للمقــاولات كان الداعــم الأكبــر والركــن الأساســي لتحقيــق مــا تربــو إليــه نفوســنا للتقــدم بشــركتنا والفــوز بحصــة اقتصاديــة فـي ً منـي شـخصيا وبنـاء كـوادر فنيـة وإدار ً يـة تسـتطيع أن تضـع بصمتهـا وترقـى بفكرهـا لمـا يخـدم هـذا الوطـن العظيـم و أيمانـا تنميــة المــوارد البشــرية وتجهيــز ك فــي تخصصــه ممــا جعلنــي أغــرس حــب العمــل والتعليــم فــي ابنائــي الأمـر الـذي جعلنـي أبعـث أبنـي المهنـدس نايـف عبـدالله رقبـان لدراسـتة الهندسـة الكهربائيـة فـي بريطانيـة وحصولـه علـى درجـة البكالريـوس والماجسـتير مـن جامعـة ادنبـره وقـد أثمـر هـذا الغـرس بتولـي الأبـن منصـب المديـر العـام لشـركة الميـال وقـد قفـزت الشـركة قفـزة رائعـة بعـد توليـه هـذا المنصـب ممـا جعلـه يتوسـع فـي انشـاء الفـروع بالريـاض والمدينـة المنـورة وينبـع ومهـد الدهـب ومكـة المكرمـة بالاضافـة الـى جـدة وهـو يتماسـى الأن مـع رؤيـة 2030التـى تخطـط لهـا المملكـة لمزيـد مـن رفـع الاقتصـاد للبـاد فـي عهـد خـادم الحرميـن الشـريفين الملـك سـلمان بـن عبدالعزيـز وسـمو ولـي عهـده صاحـب السـمو الملكـي الأميـر محمـد بـن سـلمان بـن عبدالعزيـز يحفظهـم الله

الرسالة والرؤية

الرسالة

مواكبــة أحــدث التقنيــات التــي تشــهدها الاعمــال الكهربائيــة الخاصــة لشــبكات الجهــد العالــي والمتوســط والمنخفــض والمشــاريع الفنيــة والتقنيــة علــى المســتوى الوطنــي والإقليمــي والعالمــي، ويشـمل ذلـك تطويـر وتدريـب وتأهيـل الكـوادر الفنيـة المؤهلـة، واسـتخدام أحـدث الأجهـزة والمعـدات والآليــات التــي تضمــن تقديــم أفضــل الخدمــات الفنيــة والإداريــة ومواصلــة رفــع مســتوى الأداء وفــق مــع مــا تشــهده هــذه الصناعــة مــن تطــور ً معاييــر ومؤشــرات قيــاس الأداء

الرؤية

نؤمـن تمامـا بضـرورة تحقيـق مسـتوى الجـودة الشـاملة والمتكاملـة فـي جميـع المشـار ً يع وذلـك إيمانـا بـأن ذلـك هـو السـبيل لتحقيـق النجـاح
